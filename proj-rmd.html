#Wrangle Data suitable for processing

train <- read.csv("/Users/mac/Downloads/pml-training.csv")
test <- read.csv("/Users/mac/Downloads/pml-testing.csv")

train <- train[,colMeans(is.na(train)) < .9] #removing mostly na columns
train <- train[,-c(1:7)] #removing metadata which is irrelevant to the outcome

nvz <- nearZeroVar(train)
train <- train[,-nvz]
dim(train)
## [1] 19622    53
inTrain <- createDataPartition(y=train$classe, p=0.7, list=F)
train <- train[inTrain,]
valid <- train[-inTrain,]

#Decision Tree

control <- trainControl(method="cv", number=3, verboseIter=F)

mod_trees <- train(classe~., data=train, method="rpart", trControl = control, tuneLength = 5)
fancyRpartPlot(mod_trees$finalModel)

pred_trees <- predict(mod_trees, valid)
cmtrees <- confusionMatrix(pred_trees, factor(valid$classe))
